51uvrpc_config_t* uvrpc_config_set_address(uvrpc_config_t* config, const char* address) {
52 if (!config || !address) return NULL;
53
54 if (config->address) {
55 uvrpc_free(config->address);
56 }
57
58 config->address = uvrpc_strdup(address);
59 if (!config->address) {
60 return NULL;
61 }
62
63 /* Auto-detect transport type from address prefix */
64 if (strncmp(address, "inproc://", 9) == 0) {
65 config->transport = UVRPC_TRANSPORT_INPROC;
66 } else if (strncmp(address, "ipc://", 6) == 0) {
67 config->transport = UVRPC_TRANSPORT_IPC;
68 } else if (strncmp(address, "tcp://", 6) == 0) {
69 config->transport = UVRPC_TRANSPORT_TCP;
70 } else if (strncmp(address, "udp://", 6) == 0) {
71 config->transport = UVRPC_TRANSPORT_UDP;
72 }
73 /* For addresses without prefix, keep the current/transport default */
74
75 return config;
76}
