47/* Parse address */
48static int parse_ipc_address(const char* address, char** socket_path) {
49 if (!address || !socket_path) {
50 return -1;
51 }
52
53 /* Skip protocol prefix */
54 const char* path_start = address;
55 if (strncmp(address, "ipc://", 6) == 0) {
56 path_start += 6;
57 }
58
59 *socket_path = (char*)uvrpc_alloc(strlen(path_start) + 1);
60 if (!*socket_path) {
61 return -1;
62 }
63 strcpy(*socket_path, path_start);
64
65 return 0;
66}

68/* Client read callback */
69static void on_client_read(uv_stream_t* stream, ssize_t nread, const uv_buf_t* buf) {
70 if (!stream) return;
71
72 /* Get client from stream data (pipe_handle.data should point to client, not transport) */
73 uvbus_ipc_client_t* client = (uvbus_ipc_client_t*)stream->data;
74 if (!client) return;
75
76 uvbus_transport_t* transport = (uvbus_transport_t*)client->parent_transport;
77 if (!transport) return;
78
79 if (nread < 0) {
80 if (nread != UV_EOF) {
81 if (transport->error_cb) {
82 transport->error_cb(UVBUS_ERROR_IO, uv_strerror(nread), transport->callback_ctx);
83 }
84 }
85 return;
86 }
87
88 if (nread > 0 && transport->recv_cb) {
89 /* Determine if this is server mode or client mode */
90 if (transport->is_server) {
91 /* Server mode: pass client context and server context */
92 transport->recv_cb((const uint8_t*)buf->base, nread, client, transport->callback_ctx);
93 } else {
94 /* Client mode: pass NULL for client context */
95 transport->recv_cb((const uint8_t*)buf->base, nread, NULL, transport->callback_ctx);
96 }
97 }
98}

138/* Server connection callback */
139static void on_server_connection(uv_stream_t* server, int status) {
140 uvbus_transport_t* transport = (uvbus_transport_t*)server->data;
141
142 if (status < 0) {
143 if (transport->error_cb) {
144 transport->error_cb(UVBUS_ERROR_IO, uv_strerror(status), transport->callback_ctx);
145 }
146 return;
147 }
148
149 uvbus_ipc_server_t* ipc_server = (uvbus_ipc_server_t*)transport->impl.ipc_server;
150
151 /* Create new client connection */
152 uvbus_ipc_client_t* client = (uvbus_ipc_client_t*)uvrpc_alloc(sizeof(uvbus_ipc_client_t));
153 if (!client) {
154 return;
155 }
156
157 memset(client, 0, sizeof(uvbus_ipc_client_t));
158
159 /* Initialize pipe handle */
160 uv_pipe_init(transport->loop, &client->pipe_handle, 0);
161 client->pipe_handle.data = client;
162
163 /* Accept connection */
164 if (uv_accept(server, (uv_stream_t*)&client->pipe_handle) == 0) {
165 /* Add to client list */
166 if (ipc_server->client_count >= ipc_server->client_capacity) {
167 ipc_server->client_capacity *= 2;
168 ipc_server->clients = (uvbus_ipc_client_t**)uvrpc_realloc(
169 ipc_server->clients,
170 sizeof(uvbus_ipc_client_t*) * ipc_server->client_capacity
171 );
172 }
173
174 ipc_server->clients[ipc_server->client_count++] = client;
175
176 /* Set parent reference */
177 client->parent_transport = transport;
178
179 /* Start reading */
180 uv_read_start((uv_stream_t*)&client->pipe_handle, on_client_alloc, on_client_read);
181 } else {
182 uv_close((uv_handle_t*)&client->pipe_handle, NULL);
183 uvrpc_free(client);
184 }
185}

211/* IPC listen implementation */
212static int ipc_listen(void* impl_ptr, const char* address) {
213 uvbus_transport_t* transport = (uvbus_transport_t*)impl_ptr;
214 if (!transport) {
215 return UVBUS_ERROR_INVALID_PARAM;
216 }
217
218 if (!transport->is_server) {
219 return UVBUS_ERROR_INVALID_PARAM;
220 }
221
222 /* Parse address */
223 char* socket_path = NULL;
224 if (parse_ipc_address(address, &socket_path) != 0) {
225 return UVBUS_ERROR_INVALID_PARAM;
226 }
227
228 printf("[IPC LISTEN] Starting server on %s\n", socket_path);
229 fflush(stdout);
230
231 /* Create server */
232 uvbus_ipc_server_t* server = (uvbus_ipc_server_t*)uvrpc_alloc(sizeof(uvbus_ipc_server_t));
233 if (!server) {
234 uvrpc_free(socket_path);
235 return UVBUS_ERROR_NO_MEMORY;
236 }
237
238 memset(server, 0, sizeof(uvbus_ipc_server_t));
239 server->socket_path = socket_path;
240 server->parent_transport = transport;
241 server->clients = (uvbus_ipc_client_t**)uvrpc_alloc(sizeof(uvbus_ipc_client_t*) * 10);
242 server->client_capacity = 10;
243 server->client_count = 0;
244
245 /* Initialize pipe handle */
246 uv_pipe_init(transport->loop, &server->listen_pipe, 0);
247 server->listen_pipe.data = transport;
248
249 printf("[IPC LISTEN] Pipe initialized\n");
250 fflush(stdout);
251
252 /* Bind to socket path */
253 int bind_result = uv_pipe_bind(&server->listen_pipe, socket_path);
254 if (bind_result != 0) {
255 fprintf(stderr, "[IPC LISTEN] Bind failed: %s\n", uv_strerror(bind_result));
256 uv_close((uv_handle_t*)&server->listen_pipe, NULL);
257 uvrpc_free(server->socket_path);
258 uvrpc_free(server->clients);
259 uvrpc_free(server);
260 return UVBUS_ERROR_IO;
261 }
262
263 printf("[IPC LISTEN] Bind successful\n");
264 fflush(stdout);
265
266 /* Listen */
267 int listen_result = uv_listen((uv_stream_t*)&server->listen_pipe, 128, on_server_connection);
268 if (listen_result != 0) {
269 fprintf(stderr, "[IPC LISTEN] Listen failed: %s\n", uv_strerror(listen_result));
270 uv_close((uv_handle_t*)&server->listen_pipe, NULL);
271 uvrpc_free(server->socket_path);
272 uvrpc_free(server->clients);
273 uvrpc_free(server);
274 return UVBUS_ERROR_IO;
275 }
276
277 printf("[IPC LISTEN] Listen successful\n");
278 fflush(stdout);
279
280 server->is_listening = 1;
281 transport->is_connected = 1;
282 transport->impl.ipc_server = (void*)server;
283
284 /* Set bus as active - server is ready to accept connections */
285 if (transport->parent_bus) {
286 transport->parent_bus->is_active = 1;
287 }
288
289 printf("[IPC LISTEN] Server started successfully\n");
290 fflush(stdout);
291
292 return UVBUS_OK;
293}

337/* IPC disconnect implementation */
338static void ipc_disconnect(void* impl_ptr) {
339 uvbus_transport_t* transport = (uvbus_transport_t*)impl_ptr;
340 if (!transport) {
341 return;
342 }
343
344 if (transport->is_server && transport->impl.ipc_server) {
345 uvbus_ipc_server_t* server = (uvbus_ipc_server_t*)transport->impl.ipc_server;
346 /* Close all client connections */
347 for (int i = 0; i < server->client_count; i++) {
348 if (server->clients[i]) {
349 uv_close((uv_handle_t*)&server->clients[i]->pipe_handle, NULL);
350 uvrpc_free(server->clients[i]);
351 }
352 }
353 uvrpc_free(server->clients);
354
355 /* Close listen handle */
356 uv_close((uv_handle_t*)&server->listen_pipe, NULL);
357
358 /* Remove socket file */
359 unlink(server->socket_path);
360
361 uvrpc_free(server->socket_path);
362 uvrpc_free(server);
363 transport->impl.ipc_server = NULL;
364 } else if (!transport->is_server && transport->impl.ipc_client) {
365 uvbus_ipc_client_t* client = (uvbus_ipc_client_t*)transport->impl.ipc_client;
366 /* Close client connection */
367 uv_close((uv_handle_t*)&client->pipe_handle, NULL);
368
369 uvrpc_free(client->socket_path);
370 uvrpc_free(client);
371 transport->impl.ipc_client = NULL;
372 }
373
374 transport->is_connected = 0;
375}

377/* IPC send implementation */
378static int ipc_send(void* impl_ptr, const uint8_t* data, size_t size) {
379 uvbus_transport_t* transport = (uvbus_transport_t*)impl_ptr;
380 if (!transport) {
381 return UVBUS_ERROR_INVALID_PARAM;
382 }
383
384 if (!transport->is_connected) {
385 return UVBUS_ERROR_NOT_CONNECTED;
386 }
387
388 if (transport->is_server) {
389 uvbus_ipc_server_t* server = (uvbus_ipc_server_t*)transport->impl.ipc_server;
390 /* Broadcast to all clients */
391 for (int i = 0; i < server->client_count; i++) {
392 uv_write_t* req = (uv_write_t*)uvrpc_alloc(sizeof(uv_write_t));
393 if (!req) {
394 continue;
395 }
396
397 uint8_t* data_copy = (uint8_t*)uvrpc_alloc(size);
398 if (!data_copy) {
399 uvrpc_free(req);
400 continue;
401 }
402 memcpy(data_copy, data, size);
403
404 uv_buf_t buf = uv_buf_init((char*)data_copy, size);
405 req->data = data_copy;
406
407 uv_write(req, (uv_stream_t*)&server->clients[i]->pipe_handle, &buf, 1, on_write);
408 }
409 } else {
410 uvbus_ipc_client_t* client = (uvbus_ipc_client_t*)transport->impl.ipc_client;
411 /* Send to server */
412 uv_write_t* req = (uv_write_t*)uvrpc_alloc(sizeof(uv_write_t));
413 if (!req) {
414 return UVBUS_ERROR_NO_MEMORY;
415 }
416
417 uint8_t* data_copy = (uint8_t*)uvrpc_alloc(size);
418 if (!data_copy) {
419 uvrpc_free(req);
420 return UVBUS_ERROR_NO_MEMORY;
421 }
422 memcpy(data_copy, data, size);
423
424 uv_buf_t buf = uv_buf_init((char*)data_copy, size);
425 req->data = data_copy;
426
427 if (uv_write(req, (uv_stream_t*)&client->pipe_handle, &buf, 1, on_write) != 0) {
428 uvrpc_free(data_copy);
429 uvrpc_free(req);
430 return UVBUS_ERROR_IO;
431 }
432 }
433
434 return UVBUS_OK;
435}
