39/* Transport receive callback */
40static void subscriber_recv_callback(const uint8_t* data, size_t size, void* client_ctx, void* server_ctx) {
41 uvrpc_subscriber_t* subscriber = (uvrpc_subscriber_t*)server_ctx;
42
43 if (size < 8) {
44 return;
45 }
46
47 uint8_t* p = (uint8_t*)data;
48
49 /* Parse topic length (little-endian for consistency with FlatBuffers) */
50 uint32_t topic_len = *(uint32_t*)p;
51 p += 4;
52
53 if (size < 8 + topic_len) {
54 return;
55 }
56
57 /* Extract topic */
58 char* topic = (char*)uvrpc_alloc(topic_len + 1);
59 if (!topic) {
60 return;
61 }
62 memcpy(topic, p, topic_len);
63 topic[topic_len] = '\0';
64 p += topic_len;
65
66 /* Parse data length (little-endian) */
67 uint32_t data_size = *(uint32_t*)p;
68 p += 4;
69
70 if (size < 8 + topic_len + data_size) {
71 uvrpc_free(topic);
72 return;
73 }
74
75 /* Find subscription for this topic */
76 subscription_entry_t* entry = NULL;
77 HASH_FIND_STR(subscriber->subscriptions, topic, entry);
78
79 if (entry && entry->callback) {
80 /* Call callback with the data */
81 uint8_t* data_copy = NULL;
82 if (data_size > 0) {
83 data_copy = (uint8_t*)uvrpc_alloc(data_size);
84 if (data_copy) {
85 memcpy(data_copy, p, data_size);
86 }
87 }
88
89 entry->callback(topic, data_copy, data_size, entry->ctx);
90
91 if (data_copy) uvrpc_free(data_copy);
92 }
93
94 uvrpc_free(topic);
95}

108uvrpc_subscriber_t* uvrpc_subscriber_create(uvrpc_config_t* config) {
109 if (!config || !config->loop || !config->address) return NULL;
110
111 uvrpc_subscriber_t* subscriber = uvrpc_calloc(1, sizeof(uvrpc_subscriber_t));
112 if (!subscriber) return NULL;
113
114 subscriber->loop = config->loop;
115 subscriber->address = uvrpc_strdup(config->address);
116 if (!subscriber->address) {

120
121 /* Convert uvrpc_transport_type to uvbus_transport_type_t */
122 subscriber->transport_type = (uvbus_transport_type_t)config->transport;
123 subscriber->is_connected = 0;
124 subscriber->subscriptions = NULL;
125
126 /* Create uvbus config */
127 uvbus_config_t* bus_config = uvbus_config_new();

129 uvrpc_free(subscriber->address);
130 uvrpc_free(subscriber);
131 return NULL;
132 }
133
134 uvbus_config_set_loop(bus_config, subscriber->loop);
135 uvbus_config_set_transport(bus_config, subscriber->transport_type);
136 uvbus_config_set_address(bus_config, subscriber->address);
137 uvbus_config_set_recv_callback(bus_config, subscriber_recv_callback, subscriber);
138 uvbus_config_set_connect_callback(bus_config, subscriber_connect_callback, subscriber);
139
140 /* Create uvbus client (broadcast subscriber) */
141 subscriber->uvbus = uvbus_client_new(bus_config);
142 uvbus_config_free(bus_config);
143
144 if (!subscriber->uvbus) {
145 uvrpc_free(subscriber->address);
146 uvrpc_free(subscriber);

148 }

199/* Subscribe to topic */
200int uvrpc_subscriber_subscribe(uvrpc_subscriber_t* subscriber, const char* topic,
201 uvrpc_subscribe_callback_t callback, void* ctx) {
202 if (!subscriber || !topic || !callback) return UVRPC_ERROR_INVALID_PARAM;
203
204 /* Check if already subscribed */
205 subscription_entry_t* entry = NULL;
206 HASH_FIND_STR(subscriber->subscriptions, topic, entry);
207 if (entry) return UVRPC_ERROR; /* Already subscribed */
208
209 /* Create new subscription */
210 entry = uvrpc_calloc(1, sizeof(subscription_entry_t));
211 if (!entry) return UVRPC_ERROR_NO_MEMORY;
212
213 entry->topic = uvrpc_strdup(topic);
214 if (!entry->topic) {
215 uvrpc_free(entry);
216 return UVRPC_ERROR_NO_MEMORY;
217 }
218 entry->callback = callback;
219 entry->ctx = ctx;
220
221 HASH_ADD_STR(subscriber->subscriptions, topic, entry);
222
223 printf("Subscribed to topic: %s\n", topic);
224
225 return UVRPC_OK;
226}
