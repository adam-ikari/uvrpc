225#define UVRPC_ASYNC(ctx, timeout_ms) \
226 { \
227 int _async_status = uvrpc_async_exec((ctx), (timeout_ms)); \
228 if (_async_status != UVRPC_OK) { \
229 fprintf(stderr, "Async exec failed: %d\n", _async_status); \
230 goto async_cleanup; \
231 } \
232 }

234#define UVRPC_AWAIT(call) \
235 do { \
236 int _await_status = (call); \
237 if (_await_status != UVRPC_OK) { \
238 fprintf(stderr, "Await failed: %d\n", _await_status); \
239 goto async_cleanup; \
240 } \
241 } while(0)
