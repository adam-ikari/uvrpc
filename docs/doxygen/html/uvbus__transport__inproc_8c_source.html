49/* Hash function */
50static unsigned int hash_string(const char* str) {
51 unsigned int hash = 5381;
52 int c;
53 while ((c = *str++)) {
54 hash = ((hash << 5) + hash) + c;
55 }
56 return hash % UVBUS_HASH_TABLE_SIZE;
57}

59/* Find endpoint by name */
60static inproc_endpoint_t* inproc_find_endpoint(const char* name) {
61 pthread_mutex_lock(&g_endpoint_mutex);
62 unsigned int hash = hash_string(name);
63 inproc_endpoint_t* endpoint = g_endpoint_hash[hash];
64
65 while (endpoint) {
66 if (strcmp(endpoint->name, name) == 0) {
67 pthread_mutex_unlock(&g_endpoint_mutex);
68 return endpoint;
69 }
70 endpoint = endpoint->next;
71 }
72
73 pthread_mutex_unlock(&g_endpoint_mutex);
74 return NULL;
75}

86/* Remove endpoint from hash table */
87static void inproc_remove_endpoint(inproc_endpoint_t* endpoint) {
88 pthread_mutex_lock(&g_endpoint_mutex);
89 unsigned int hash = hash_string(endpoint->name);
90 inproc_endpoint_t** ptr = &g_endpoint_hash[hash];

92 while (*ptr) {

103/* Add client to endpoint */
104static void inproc_add_client(inproc_endpoint_t* endpoint, void* client) {
105 pthread_mutex_lock(&g_endpoint_mutex);
106 if (endpoint->client_count >= endpoint->client_capacity) {
107 endpoint->client_capacity *= 2;
108 endpoint->clients = (void**)uvrpc_realloc(
109 endpoint->clients,
110 sizeof(void*) * endpoint->client_capacity
111 );
112 }
113
114 endpoint->clients[endpoint->client_count++] = client;
115 pthread_mutex_unlock(&g_endpoint_mutex);
116}

118/* Remove client from endpoint */
119static void inproc_remove_client(inproc_endpoint_t* endpoint, void* client) {
120 pthread_mutex_lock(&g_endpoint_mutex);
121 for (int i = 0; i < endpoint->client_count; i++) {
122 if (endpoint->clients[i] == client) {
123 /* Shift remaining clients */
124 for (int j = i; j < endpoint->client_count - 1; j++) {
125 endpoint->clients[j] = endpoint->clients[j + 1];
126 }
127 endpoint->client_count--;
128 pthread_mutex_unlock(&g_endpoint_mutex);
129 return;
130 }
131 }
132 pthread_mutex_unlock(&g_endpoint_mutex);
133}

135/* Send to all clients */
136static void inproc_send_to_all(inproc_endpoint_t* endpoint,
137 const uint8_t* data, size_t size) {
138 pthread_mutex_lock(&g_endpoint_mutex);
139 /* Copy client count to avoid holding lock during callbacks */
140 int client_count = endpoint->client_count;
141 void** clients = (void**)uvrpc_alloc(sizeof(void*) * client_count);
142 if (clients) {
143 memcpy(clients, endpoint->clients, sizeof(void*) * client_count);
144 }
145 pthread_mutex_unlock(&g_endpoint_mutex);
146
147 if (!clients) return;
148
149 /* Trigger callbacks without holding the lock */
150 for (int i = 0; i < client_count; i++) {
151 inproc_client_t* client = (inproc_client_t*)clients[i];
152 if (client && client->is_active) {
153 /* Use client's own callback instead of endpoint's */
154 if (client->recv_cb) {
155 /* Pass client as client_ctx, and server context (if any) as server_ctx */
156 /* For INPROC, client->callback_ctx is the server context */
157 client->recv_cb(data, size, client, client->callback_ctx);
158 }
159 }
160 }
161
162 uvrpc_free(clients);
163}

183/* INPROC listen implementation */
184static int inproc_listen(void* impl_ptr, const char* address) {
185 uvbus_transport_t* transport = (uvbus_transport_t*)impl_ptr;
186 if (!transport) {
187 return UVBUS_ERROR_INVALID_PARAM;
188 }
189
190 if (!transport->is_server) {
191 return UVBUS_ERROR_INVALID_PARAM;
192 }
193
194 /* Skip protocol prefix */
195 const char* name = address;
196 if (strncmp(address, "inproc://", 9) == 0) {
197 name += 9;
198 }
199
200 /* Check if endpoint already exists */
201 inproc_endpoint_t* endpoint = inproc_find_endpoint(name);
202 if (endpoint) {
203 return UVBUS_ERROR_ALREADY_EXISTS;
204 }
205
206 /* Create endpoint */
207 endpoint = (inproc_endpoint_t*)uvrpc_alloc(sizeof(inproc_endpoint_t));
208 if (!endpoint) {
209 return UVBUS_ERROR_NO_MEMORY;
210 }
211
212 endpoint->name = uvrpc_strdup(name);
213 endpoint->server_transport = transport;
214 endpoint->clients = (void**)uvrpc_alloc(sizeof(void*) * UVBUS_INITIAL_CLIENT_CAPACITY);
215 endpoint->client_capacity = UVBUS_INITIAL_CLIENT_CAPACITY;
216 endpoint->client_count = 0;
217 endpoint->next = NULL;
218
219 /* Set callbacks */
220 endpoint->recv_cb = transport->recv_cb;
221 endpoint->error_cb = transport->error_cb;
222 endpoint->callback_ctx = transport->callback_ctx;
223
224 /* Add to global list */
225 inproc_add_endpoint(endpoint);
226
227 transport->impl.inproc_server = (void*)endpoint;
228 transport->is_connected = 1;
229
230 /* Set bus as active */
231 if (transport->parent_bus) {
232 transport->parent_bus->is_active = 1;
233 }
234
235 return UVBUS_OK;
236}

302/* INPROC disconnect implementation */
303static void inproc_disconnect(void* impl_ptr) {
304 uvbus_transport_t* transport = (uvbus_transport_t*)impl_ptr;
305 if (!transport) {
306 return;
307 }
308
309 if (transport->is_server && transport->impl.inproc_server) {
310 inproc_endpoint_t* endpoint = (inproc_endpoint_t*)transport->impl.inproc_server;
311 /* Remove endpoint from global list */
312 inproc_remove_endpoint(endpoint);
313
314 /* Free all clients */
315 for (int i = 0; i < endpoint->client_count; i++) {
316 if (endpoint->clients[i]) {
317 inproc_client_t* client = (inproc_client_t*)endpoint->clients[i];
318 client->is_active = 0;
319 }
320 }
321
322 /* Free endpoint */
323 uvrpc_free(endpoint->name);
324 uvrpc_free(endpoint->clients);
325 uvrpc_free(endpoint);
326 transport->impl.inproc_server = NULL;
327 } else if (!transport->is_server && transport->impl.inproc_client) {
328 inproc_client_t* client = (inproc_client_t*)transport->impl.inproc_client;
329 /* Remove from endpoint */
330 if (client->server_endpoint) {
331 inproc_remove_client(client->server_endpoint, client);
332 }
333
334 client->is_active = 0;
335 uvrpc_free(client);
336 transport->impl.inproc_client = NULL;
337 }
338
339 transport->is_connected = 0;
340}

342/* INPROC send implementation */
343static int inproc_send(void* impl_ptr, const uint8_t* data, size_t size) {
344 uvbus_transport_t* transport = (uvbus_transport_t*)impl_ptr;
345 if (!transport) {
346 return UVBUS_ERROR_INVALID_PARAM;
347 }
348
349 if (!transport->is_connected) {
350 return UVBUS_ERROR_NOT_CONNECTED;
351 }
352
353 if (transport->is_server && transport->impl.inproc_server) {
354 inproc_endpoint_t* endpoint = (inproc_endpoint_t*)transport->impl.inproc_server;
355 /* Broadcast to all clients */
356 inproc_send_to_all(endpoint, data, size);
357 } else if (!transport->is_server && transport->impl.inproc_client) {
358 inproc_client_t* client = (inproc_client_t*)transport->impl.inproc_client;
359 if (client->server_endpoint) {
360 /* Send to server */
361 inproc_endpoint_t* endpoint = client->server_endpoint;
362 if (endpoint->recv_cb) {
363 /* Pass client as client_ctx, and server context as server_ctx */
364 endpoint->recv_cb(data, size, client, endpoint->callback_ctx);
365 }
366 }
367 }
368
369 return UVBUS_OK;
370}
