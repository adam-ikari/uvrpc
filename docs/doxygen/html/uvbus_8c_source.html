118uvbus_t* uvbus_server_new(uvbus_config_t* config) {
119 if (!config || !config->loop) {
120 return NULL;
121 }
122
123 uvbus_t* bus = (uvbus_t*)uvrpc_alloc(sizeof(uvbus_t));
124 if (!bus) {
125 return NULL;
126 }
127
128 memset(bus, 0, sizeof(uvbus_t));
129 bus->config = *config;
130 bus->is_active = 0;
131
132 /* Create transport */
133 bus->transport = create_transport(config->transport, config->loop);
134 if (!bus->transport) {
135 uvrpc_free(bus);
136 return NULL;
137 }
138
139 bus->transport->parent_bus = bus;
140 bus->transport->is_server = 1;
141 bus->transport->type = config->transport;
142 bus->transport->loop = config->loop;
143
144 if (config->address) {
145 bus->transport->address = uvrpc_strdup(config->address);
146 }
147
148 /* Set callbacks */
149 bus->transport->recv_cb = config->recv_cb;
150 bus->transport->connect_cb = config->connect_cb;
151 bus->transport->close_cb = config->close_cb;
152 bus->transport->error_cb = config->error_cb;
153 bus->transport->callback_ctx = config->callback_ctx;
154
155 return bus;

198uvbus_error_t uvbus_listen(uvbus_t* bus) {
199 if (!bus || !bus->transport || !bus->transport->vtable) {
200 return UVBUS_ERROR_INVALID_PARAM;
201 }
202
203 if (!bus->transport->is_server) {
204 return UVBUS_ERROR_INVALID_PARAM;
205 }
206
207 if (bus->transport->vtable->listen) {
208 uvbus_error_t result = bus->transport->vtable->listen(bus->transport, bus->transport->address);
209 if (result == UVBUS_OK) {
210 bus->is_active = 1;
211 }
212 return result;
213 }
214
215 return UVBUS_ERROR;
216}

345void uvbus_free(uvbus_t* bus) {
346 if (!bus) {
347 return;
348 }
349
350 /* Stop if active */
351 if (bus->is_active) {
352 if (bus->transport && bus->transport->is_server) {
353 uvbus_stop(bus);
354 } else {
355 uvbus_disconnect(bus);
356 }
357 }
358
359 /* Free transport */
360 if (bus->transport) {
361 if (bus->transport->vtable && bus->transport->vtable->free) {
362 bus->transport->vtable->free(bus->transport);
363 } else {
364 if (bus->transport->address) {
365 uvrpc_free(bus->transport->address);
366 }
367 uvrpc_free(bus->transport);
368 }
369 }
370
371 uvrpc_free(bus);
