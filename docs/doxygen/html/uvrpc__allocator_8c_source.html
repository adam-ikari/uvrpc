77 g_allocator_type = type;
78
79 if (type == UVRPC_ALLOCATOR_CUSTOM && custom) {
80 if (custom->alloc && custom->calloc && custom->realloc && custom->free) {
81 g_custom_allocator = *custom;

170void uvrpc_free(void* ptr) {
171 if (!ptr) return;
172
173 switch (g_allocator_type) {
174 case UVRPC_ALLOCATOR_SYSTEM:
175 system_free(ptr);
176 break;
177 case UVRPC_ALLOCATOR_MIMALLOC:
178#if UVRPC_DEFAULT_ALLOCATOR == UVRPC_ALLOCATOR_MIMALLOC
179 mimalloc_free(ptr);
180#else
181 fprintf(stderr, "Error: Mimalloc not compiled in\n");
182 system_free(ptr);
183#endif
184 break;
185 case UVRPC_ALLOCATOR_CUSTOM:
186 if (g_custom_allocator.free) {
187 g_custom_allocator.free(ptr);
188 } else {
189 system_free(ptr);
190 }
191 break;
192 default:
193 system_free(ptr);
194 break;
195 }
