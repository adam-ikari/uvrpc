75 if (!publisher || !publisher->uvbus) return UVRPC_ERROR_INVALID_PARAM;

110/* Publish message */
111int uvrpc_publisher_publish(uvrpc_publisher_t* publisher, const char* topic,
112 const uint8_t* data, size_t size,
113 uvrpc_publish_callback_t callback, void* ctx) {
114 if (!publisher || !topic) return UVRPC_ERROR_INVALID_PARAM;
115
116 if (!publisher->is_running) {
117 return UVRPC_ERROR;
118 }
119
120 /* Encode broadcast message: [topic_len, topic, data_len, data] */
121 size_t topic_len = strlen(topic);
122 size_t total_size = 4 + topic_len + 4 + size;
123
124 uint8_t* msg = (uint8_t*)uvrpc_alloc(total_size);
125 if (!msg) return UVRPC_ERROR_NO_MEMORY;
126
127 uint8_t* p = msg;
128
129 /* Topic length (4 bytes little-endian for consistency with FlatBuffers) */
130 *(uint32_t*)p = (uint32_t)topic_len;
131 p += 4;
132
133 /* Topic */
134 memcpy(p, topic, topic_len);
135 p += topic_len;
136
137 /* Data length (4 bytes little-endian) */
138 *(uint32_t*)p = (uint32_t)size;
139 p += 4;
140
141 /* Data */
142 if (data && size > 0) {
143 memcpy(p, data, size);
144 }
145
146 /* Send message */
147 if (publisher->uvbus) {
148 uvbus_client_send(publisher->uvbus, msg, total_size);
149 }
150
151 uvrpc_free(msg);
152
153 /* Call callback */
154 if (callback) {
155 callback(UVRPC_OK, ctx);
156 }
157
158 return UVRPC_OK;
159}
